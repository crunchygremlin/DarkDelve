### DynamicDifficultyService (domain service)
```python
class DynamicDifficultyService:
    """Service for dynamically adjusting difficulty based on player stats and DM LLM evaluation."""
    
    def __init__(
        self,
        dungeon_master_service: 'DungeonMasterService',
        llm_worker: 'LLMWorker',
        player_profile_service: 'PlayerProfileService'
    ):
        self.dungeon_master_service = dungeon_master_service
        self.llm_worker = llm_worker
        self.player_profile_service = player_profile_service
        self.last_evaluated_level = 0
    
    def evaluate_and_adjust_difficulty(
        self,
        player_entity: 'Entity',
        current_level: int
    ) -> DifficultyAdjustment:
        """
        Evaluate player stats using DM LLM and return difficulty adjustment.
        
        Args:
            player_entity: The player entity to evaluate
            current_level: The current level the player is on
            
        Returns:
            DifficultyAdjustment object containing recommended changes
        """
        # Check if we need to evaluate (level change)
        if current_level <= self.last_evaluated_level:
            return DifficultyAdjustment.no_change()
        
        # Get player's max stats
        player_stats = self._get_player_max_stats(player_entity)
        
        # Request evaluation from DM LLM
        llm_response = self.llm_worker.evaluate_player_stats(
            player_stats=player_stats,
            current_level=current_level
        )
        
        # Parse LLM response into difficulty adjustment
        adjustment = self._parse_llm_response(llm_response)
        
        # Update tracking
        self.last_evaluated_level = current_level
        
        return adjustment
    
    def _get_player_max_stats(self, player_entity: 'Entity') -> Dict[str, int]:
        """Extract maximum stats from player entity."""
        stats = {}
        # Get base stats from player components
        if hasattr(player_entity, 'fighter'):
            stats['health'] = player_entity.fighter.max_hp
            stats['attack'] = player_entity.fighter.power
            stats['defense'] = player_entity.fighter.defense
        if hasattr(player_entity, 'power'):
            stats['power_level'] = player_entity.power.level
        return stats
    
    def _parse_llm_response(self, llm_response: Dict[str, Any]) -> DifficultyAdjustment:
        """Parse LLM response into difficulty adjustment."""
        # Default to no change
        adjustment = DifficultyAdjustment.no_change()
        
        # Extract adjustment factors from LLM response
        if 'difficulty_modifier' in llm_response:
            modifier = llm_response['difficulty_modifier']
            adjustment.spawn_rate_modifier = modifier
            adjustment.monster_health_modifier = modifier
            adjustment.monster_damage_modifier = modifier
        
        if 'specific_adjustments' in llm_response:
            adjustments = llm_response['specific_adjustments']
            if 'spawn_rate' in adjustments:
                adjustment.spawn_rate_modifier = adjustments['spawn_rate']
            if 'monster_health' in adjustments:
                adjustment.monster_health_modifier = adjustments['monster_health']
            if 'monster_damage' in adjustments:
                adjustment.monster_damage_modifier = adjustments['monster_damage']
        
        return adjustment
```

### Application Service DynamicDifficultyService
```python
class ApplicationDynamicDifficultyService:
    """Application service that coordinates dynamic difficulty adjustment."""
    
    def __init__(
        self,
        domain_service: 'DynamicDifficultyService',
        event_bus: 'EventBus',
        game_session: 'GameSession'
    ):
        self.domain_service = domain_service
        self.event_bus = event_bus
        self.game_session = game_session
        self._register_event_handlers()
    
    def _register_event_handlers(self):
        """Register for level change events."""
        self.event_bus.subscribe(
            LevelChangedEvent,
            self._on_level_changed
        )
    
    def _on_level_changed(self, event: 'LevelChangedEvent'):
        """Handle level change events by triggering difficulty evaluation."""
        player_entity = self.game_session.player_entity
        if player_entity is None:
            return
        
        # Evaluate and adjust difficulty
        adjustment = self.domain_service.evaluate_and_adjust_difficulty(
            player_entity=player_entity,
            current_level=event.new_level
        )
        
        # Apply adjustment to level generation services
        self._apply_difficulty_adjustment(adjustment, event.new_level)
    
    def _apply_difficulty_adjustment(
        self,
        adjustment: DifficultyAdjustment,
        level_number: int
    ):
        """Apply difficulty adjustment to level generation services."""
        # Notify level generators of the adjustment
        self.event_bus.publish(
            DifficultyAdjustedEvent(
                level_number=level_number,
                adjustment=adjustment
            )
        )
```

```python
class LLMWorker:
    """Extended LLM worker for player stats evaluation."""
    
    def evaluate_player_stats(
        self,
        player_stats: Dict[str, int],
        current_level: int
    ) -> Dict[str, Any]:
        """
        Evaluate player stats using DM LLM and return difficulty adjustment recommendations.
        
        Args:
            player_stats: Dictionary of player stats (health, attack, defense, etc.)
            current_level: Current level the player is on
            
        Returns:
            Dictionary containing difficulty adjustment recommendations
        """
        # Construct prompt for DM LLM
        prompt = self._build_evaluation_prompt(player_stats, current_level)
        
        # Query LLM (no caching as per requirements)
        response = self._query_llm(prompt, use_cache=False)
        
        # Parse and return response
        return self._parse_evaluation_response(response)
    
    def _build_evaluation_prompt(
        self,

        current_level: int
    ) -> str:
        """Build prompt for DM LLM to evaluate player stats."""
        return f"""
        As the Dungeon Master, evaluate the player's current stats and recommend 
        difficulty adjustments for the next level.
        
        Player Stats:
        - Health: {player_stats.get('health', 0)}
        - Attack: {player_stats.get('attack', 0)}
        - Defense: {player_stats.get('defense', 0)}
        - Power Level: {player_stats.get('power_level', 0)}
        - Current Level: {current_level}
        
        Provide your assessment in JSON format with the following structure:
        {{
            "difficulty_modifier": float,  // Overall difficulty multiplier (0.5 to 2.0)
            "specific_adjustments": {{
                "spawn_rate": float,       // Monster spawn rate multiplier (0.5 to 2.0)
```

```python
        )
         
        # Spawn guard patrols with adjusted counts

        # ... similar adjustments for den creatures, roaming creatures, etc.

        self,
        base_count_func: Callable[[], int],
        modifier: float

        adjusted_count = max(1, int(base_count * modifier))
        return adjusted_count
```

### Data Classes
```python
@dataclass
class DifficultyAdjustment:
    """Represents adjustments to difficulty parameters."""
    spawn_rate_modifier: float = 1.0
    monster_health_modifier: float = 1.0
    monster_damage_modifier: float = 1.0
    experience_reward_modifier: float = 1.0
    loot_quality_modifier: float = 1.0
    
    @classmethod
    def no_change(cls) -> 'DifficultyAdjustment':
        """Create an adjustment that maintains current difficulty."""
        return cls(
            spawn_rate_modifier=1.0,
            monster_health_modifier=1.0,
            monster_damage_modifier=1.0,
            experience_reward_modifier=1.0,
            loot_quality_modifier=1.0
        )
    
    def is_significant_change(self, threshold: float = 0.1) -> bool:
        """Check if this adjustment represents a significant change from normal."""
        return (
            abs(self.spawn_rate_modifier - 1.0) > threshold or
            abs(self.monster_health_modifier - 1.0) > threshold or
            abs(self.monster_damage_modifier - 1.0) > threshold
        )

@dataclass
class LevelChangedEvent:
    """Event published when player changes levels."""
    old_level: int
    new_level: int
    player_entity: Optional['Entity'] = None

@dataclass  
class DifficultyAdjustedEvent:
    """Event published when difficulty has been adjusted."""
    level_number: int
    adjustment: DifficultyAdjustment
```

## Integration Notes

1. **Integration with Existing Systems**:
   - The `DynamicDifficultyService` integrates with the existing `DungeonMasterService` through dependency injection
   - The application service subscribes to `LevelChangedEvent` from the existing event system
   - Modifications to `Floor1Generator` and `Floor1Spawner` accept optional `DifficultyAdjustment` parameters
   - The `LLMWorker` extension integrates with existing Ollama service infrastructure

2. **Data Flow**:
   - Player changes level → `LevelChangedEvent` published
   - `SystemHandler` receives event and delegates to `ApplicationDynamicDifficultyService`
   - Application service calls domain service to evaluate player stats
   - Domain service uses `LLMWorker` to get DM LLM evaluation (no caching)
   - Domain service returns `DifficultyAdjustment`
   - Application service publishes `DifficultyAdjustedEvent`
   - Level generators and spawners receive adjusted parameters through constructor injection
   - Generation/spawning algorithms use adjusted parameters to modify monster generation

3. **Integration Points**:
   - Event system: Subscribes to `LevelChangedEvent`, publishes `DifficultyAdjustedEvent`
   - Level generation: Modified constructors accept difficulty adjustments
   - Entity spawning: Modified constructors accept difficulty adjustments
   - LLM services: Extended `LLMWorker` with player stats evaluation method
   - Dependency injection: Services receive dependencies through constructors

## Performance Considerations

1. **LLM Call Frequency**: 
   - LLM is called only on level changes, not every frame or action
   - Explicitly configured to not cache results as per requirements
   - Minimal performance impact as level changes are infrequent

2. **Memory Usage**:
   - Only stores last evaluated level integer and adjustment objects
   - No caching of LLM responses or player histories
   - Adjustment objects are lightweight data classes

3. **Computation Overhead**:
   - Minimal computation in stat extraction and adjustment application
   - Most expensive operation is the LLM call, which happens infrequently
   - Adjustment application involves simple multiplication of integers/floats

4. **Scalability**:
   - System scales linearly with number of level transitions
   - No persistent state that grows over time
   - Easy to disable by setting adjustment multipliers to 1.0

## Error Handling and Edge Cases

1. **LLM Failure Handling**:
   - If LLM query fails or returns invalid JSON, returns default adjustment (1.0 for all modifiers)
   - Logging of LLM failures for debugging (would be implemented in actual code)
   - Game continues with normal difficulty when LLM is unavailable

2. **Edge Cases**:
   - **Starting Level (Level 0)**: First level change from 0→1 triggers evaluation
   - **Rapid Level Changes**: Only evaluates when level increases beyond last evaluated level
   - **Level Decreases**: No evaluation when player goes back to previous levels (e.g., using stairs up)
   - **Player Death**: Handled by game systems; no special handling needed in difficulty system
   - **Null Player Entity**: Gracefully handles case where player entity is not available
   - **Extreme Stat Values**: Adjustment multipliers are clamped implicitly by game balance systems
   - **Missing Stats**: Gracefully handles missing player components with default values

3. **Validation**:
   - Adjustment values are validated by receiving systems (generators, spawners)
   - Extreme values would be handled by existing game balance systems
   - No validation needed in difficulty service as it only provides recommendations

4. **Fallback Behavior**:
   - Default to no change (multiplier of 1.0) when uncertain
   - Systems continue to function normally with default difficulty
   - Clear logging would help diagnose issues in production

This design satisfies all requirements:
- Uses DM LLM to evaluate player stats at each level change
- Adjusts monster generation parameters (spawn rates, types, difficulty)
- Does not cache LLM judgments (explicitly set use_cache=False)
- Integrates with existing level generation (floor1_generator.py) and monster spawning (floor1_spawner.py)
- Uses existing DM LLM infrastructure (llm_worker.py, dungeon_master_service.py)
- Works with existing game loop and event system
- Follows existing code patterns (dependency injection, event-driven architecture)
- Includes comprehensive error handling and edge case consideration
- Minimal performance impact
- Maintains compatibility with save/load systems (no persistent state added)
