def _register_handlers(self):
    """Register event handlers for level change events."""
    from src.application.event_system.base_event import EventHandler
    
    # Create a handler for level change events
    class LevelChangeHandler(EventHandler):
        def __init__(self, system_handler_instance):
            super().__init__("system_handler_level_change", ["level_change"])
            self.system_handler = system_handler_instance
        
        def handle_event(self, event) -> bool:
            # Expect event.data to have 'old_level' and 'new_level'
            old_level = event.data.get('old_level')
            new_level = event.data.get('new_level')
            if old_level is not None and new_level is not None:
                self.system_handler.handle_level_change(old_level, new_level)
                return True
            return False
    
    # Register the handler with the event bus
    level_change_handler = LevelChangeHandler(self)
    self.event_bus.register_handler(level_change_handler)
```

def _register_event_handlers(self):
    """Register for level change events and difficulty adjusted events."""
    from src.application.event_system.base_event import EventHandler
    
    # Handler for level change events
    class LevelChangeHandler(EventHandler):
        def __init__(self, service_instance):
            super().__init__("difficulty_service_level_change", ["level_change"])
            self.service = service_instance
        
        def handle_event(self, event) -> bool:
            old_level = event.data.get('old_level')
            new_level = event.data.get('new_level')
            if old_level is not None and new_level is not None:
                self.service.handle_level_change(old_level, new_level)
                return True
            return False
    
    # Handler for difficulty adjusted events (for logging/UI updates)
    class DifficultyAdjustedHandler(EventHandler):
        def __init__(self, service_instance):
            super().__init__("difficulty_service_difficulty_adjusted", ["difficulty_adjusted"])
            self.service = service_instance
        
        def handle_event(self, event) -> bool:
            # Log the difficulty adjustment for debugging/UI
            level_number = event.data.get('level_number')
            adjustment = event.data.get('adjustment')
            if level_number is not None and adjustment is not None:
                # Example logging - can be extended for UI updates, achievements, etc.
                import logging
                logger = logging.getLogger(__name__)
                logger.info(f"Difficulty adjusted for level {level_number}: {adjustment}")
                return True
            return False
    
    # Register handlers
    level_change_handler = LevelChangeHandler(self)
    difficulty_adjusted_handler = DifficultyAdjustedHandler(self)
    self.event_bus.register_handler(level_change_handler)
    self.event_bus.register_handler(difficulty_adjusted_handler)
```

# Integration Notes
1. SystemHandler will register a LevelChangeHandler with the event bus to listen for "level_change" events. When such an event is received, it will call its handle_level_change method, which in turn delegates to the ApplicationDynamicDifficultyService.
2. ApplicationDynamicDifficultyService will register two handlers:
   - LevelChangeHandler: listens for "level_change" events and calls the service's handle_level_change method.
   - DifficultyAdjustedHandler: listens for "difficulty_adjusted" events and logs them (can be extended for UI updates, achievements, etc.).
3. Level change events will be published by the GameSession when the player's level changes (e.g., when advancing to a new floor). The event will include the old level, new level, and the player entity.
4. The difficulty_adjusted event is already published by the ApplicationDynamicDifficultyService when it applies an adjustment. The new DifficultyAdjustedHandler will log these events, making the event useful for debugging and potential UI updates.
5. The floor1_generator and floor1_spawner do not need to publish level change events directly; the GameSession is the central authority for level progression and will publish the event.

# Risks and Mitigations
- Risk: Circular dependency between SystemHandler and ApplicationDynamicDifficultyService.
  Mitigation: We are passing the service to the handler via constructor, and the service already has a reference to the event bus. No direct circular dependency is created.
  
- Risk: Multiple handlers for the same event causing duplicate processing.
  Mitigation: Each handler has a unique ID, and the event bus will call all registered handlers. This is acceptable as long as each handler performs a distinct function (logging vs. processing).
  
- Risk: Memory leaks if handlers are not unregistered.
  Mitigation: The current system does not implement handler unregistration, but the lifecycle of these handlers is tied to the lifetime of the SystemHandler and ApplicationDynamicDifficultyService objects, which are long-lived for the game session.
  
- Risk: Level change events may be published multiple times for the same level change.
  Mitigation: Ensure that the level change event is published exactly once per level transition in the GameSession.
  
- Risk: The event bus may not be injected correctly in all places.
  Mitigation: We will ensure the GameSession has an event_bus attribute set during initialization or via dependency injection.
