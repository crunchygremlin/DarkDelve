def generate_floor1_description_async(self, game, state, request_queue):
    """Enqueue a floor 1 description request to the LLM worker queue.
    
    Returns a procedural fallback description immediately.
    The async response will be delivered to llm_response_queue later.
    """
    import time
    player_class = getattr(game.player, "player_class", "adventurer")
    player_hp = game.player.max_hp
    theme = getattr(game, "current_theme", None)
    theme_name = theme.name if theme else "dungeon entrance"
    theme_desc = theme.description if theme else "A cold stone dungeon"

    prompt = f"""You are a dungeon master describing the entrance level of a roguelike dungeon.

Player: {player_class} (HP: {player_hp})
Theme: {theme_name} - {theme_desc}
Depth: 1 (entrance level)

Describe this floor in 3-5 sentences. Include:
- The entrance area (top of the map)
- Corridors and side rooms
- Monster types present (e.g., spiders, guards, rats, bats, trolls)
- Special features (dens, patrol routes, corpses, stairs)
- The stairs down (bottom of the map)

Use concrete monster names: spider, giant_spider, guard, dungeon_guard, sergeant, rat, cave_rat, bat, cave_bat, troll, troll_scavenger, fungal_creeper.
Use concrete feature names: corridor, den, patrol, corpse, stairs, web, nest.

Return ONLY the description text, no JSON, no markdown."""

    cache_key = f"floor1_desc_{theme_name}_{player_class}"

    # Check cache first — synchronous cache read is safe (SQLite, no network)
    if self.llm_config.get("cache_enabled", True):
        cached = self.cache.get(cache_key, prompt)
        if cached:
            # Put cached result directly into response queue as if LLM returned it
            request_queue.put({
                "type": "floor1_description",
                "response": cached,
                "cache_key": cache_key,
                "prompt": prompt,
                "turn_number": getattr(state, "turn", 0),
                "prompt_summary": f"Floor 1 description for {player_class}",
            })
            return cached

    # Enqueue the request for the LLM worker thread
    request_queue.put({
        "type": "floor1_description",
        "prompt": prompt,
        "cache_key": cache_key,
        "theme_name": theme_name,
        "player_class": player_class,
        "turn_number": getattr(state, "turn", 0),
        "prompt_summary": f"Floor 1 description for {player_class}",
    })

    # Return procedural fallback immediately (never blocks)
    fallback = (
        f"The {theme_name} stretches before you. "
        f"Stone corridors branch into shadowed side rooms where spiders and rats make their dens. "
        f"Dungeon guards patrol the main path in pairs, their sergeant watching from a side alcove. "
        f"A few corpses lie scattered, picked clean by scavengers. "
        f"The stairs down wait at the far end, half-hidden by webs."
    )
    return fallback
```

def _process_llm_responses(self):
    """Process LLM responses from the response queue."""
    while not self.llm_response_queue.empty():
        try:
            result = self.llm_response_queue.get_nowait()
            response_type = result.get('type', 'behavior')
            
            # Handle floor1_description responses
            if response_type == 'floor1_description':
                description = result.get('description', '')
                if description:
                    self.floor1_description = description
                    self.add_message(f"The dungeon master whispers: {description[:100]}...")
                    if hasattr(self, 'llm_logger') and self.llm_logger:
                        from src.domain.value_objects.llm_logging import LLMCallLog
                        self.llm_logger.log_call(LLMCallLog(
                            call_id=f"floor1_desc_{time.time()}",
                            timestamp=time.time(),
                            context="floor1_description",
                            entity_id=None,
                            prompt_summary=result.get('prompt_summary', ''),
                            response_summary=description[:200],
                            latency_ms=result.get('latency_ms', 0),
                            tokens_used=0,
                            success=True,
                            turn_number=result.get('turn_number', 0),
                            call_type='floor1_description',
                        ))
                continue
            
            # Handle behavior responses (existing code)
            entity_id = result.get('entity_id')
            for entity in self.entities:
                if getattr(entity, 'id', None) == entity_id:
                    if result.get('success') and result.get('behavior_script'):
                        entity.behavior_script = result['behavior_script']
                    break
        except Exception:
            break

import signal

def test_game_state_serialization(self):
    """Test game state serialization with timeout to catch hangs."""
    
    def timeout_handler(signum, frame):
        raise TimeoutError("test_game_state_serialization hung — likely blocking on LLM call")
    
    # Set 10-second timeout for this test
    old_handler = signal.signal(signal.SIGALRM, timeout_handler)
    signal.alarm(10)
    
    try:
        self.game.create_player()
        self.game.generate_level(1, "main")

        # Modify game state
        self.game.state.depth = 3
        self.game.state.branch = "catacombs"
        self.game.state.kills = 5

        # Serialize game state
        serialized = {
            "run_id": self.game.state.run_id,
            "depth": self.game.state.depth,
            "branch": self.game.state.branch,
            "kills": self.game.state.kills,
            "flags": list(self.game.state.flags)
        }

        # Check serialization
        self.assertEqual(serialized["depth"], 3)
        self.assertEqual(serialized["branch"], "catacombs")
        self.assertEqual(serialized["kills"], 5)
        self.assertIn("run_id", serialized)
    finally:
        signal.alarm(0)
        signal.signal(signal.SIGALRM, old_handler)
```

## 7. Integration Notes

- The existing `llm_request_queue` / `llm_response_queue` pattern is reused exactly as-is. The worker thread (`llm_worker_func`) already handles `behavior` and `level_design` types; we add `floor1_description` as a third type.
- The `Game._process_llm_responses()` method already runs every turn in `main_loop()`. It will pick up floor1_description responses without any additional wiring.
- The `generate_floor1_description_fallback()` method is pure Python with no I/O — safe to call from any context.
- The `generate_floor1_description_async()` method only enqueues to a `queue.Queue` (thread-safe, non-blocking) and returns immediately.
- The `dm_enabled` flag controls whether the LLM path is attempted. When `False`, the fallback is used directly.
- The `Game.floor1_description` attribute stores the latest LLM-provided description for use in rendering or logging.

## 8. Risks and Mitigations

| Risk | Mitigation |
|------|------------|
| Worker thread not started when `dm_enabled=False` | The `generate_floor1_description_fallback()` path is used when `dm_enabled=False` or queue is None — no worker needed |
| Queue full (backpressure) | `queue.Queue.put()` blocks only if queue is full; default maxsize=50 is sufficient. Use `put_nowait()` and catch `queue.Full` if needed |
| Cache read is slow | SQLite reads are sub-millisecond; safe to call synchronously |
| Test timeout on slow CI | Signal-based timeout in test catches hangs early with clear error message |
| Fallback description is static | Acceptable — it's only used when LLM is unavailable. When LLM is available, the async response replaces it |
| `llm_worker_func` doesn't have access to `ollama_service` | `DungeonMasterAgent` already holds `self.ollama_service` (set in `Game.initialize()`). The worker receives `dm_agent` which has this attribute |
