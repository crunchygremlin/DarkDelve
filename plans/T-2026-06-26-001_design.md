```python
def __str__(self, perspective: str = "neutral") -> str:
    """
    Generate a human-readable combat message.
    
    Args:
        perspective: One of "attacker_is_player", "defender_is_player", "neutral"
            - "attacker_is_player": Player is the attacker (first-person language)
            - "defender_is_player": Player is the defender (second-person language)
            - "neutral": Neither is player (third-person language)
    """
    if getattr(self, 'out_of_range', False):
        if perspective == "attacker_is_player":
            return f"Player is out of range to attack {self.defender_name}!"
        elif perspective == "defender_is_player":
            return f"{self.attacker_name} is out of range to attack you!"
        else:
            return f"{self.attacker_name} is out of range to attack {self.defender_name}!"
    
    roll_text = f"[Roll: {self.total_roll} vs AC {self.target_ac}]"
    
    if self.result == HitResult.CRITICAL:
        if perspective == "attacker_is_player":
            return f"Player strikes {self.defender_name} critically! {roll_text} CRITICAL HIT! Damage: {self.damage}"
        elif perspective == "defender_is_player":
            return f"{self.attacker_name} lands a critical hit on you! {roll_text} CRITICAL HIT! Damage: {self.damage}"
        else:
            return f"{self.attacker_name} strikes {self.defender_name} critically! {roll_text} CRITICAL HIT! Damage: {self.damage}"
    
    elif self.result == HitResult.HIT:
        if perspective == "attacker_is_player":
            return f"Player attacks {self.defender_name}! {roll_text} HIT! Damage: {self.damage}"
        elif perspective == "defender_is_player":
            return f"{self.attacker_name} attacks player! {roll_text} HIT! Damage: {self.damage}"
        else:
            return f"{self.attacker_name} attacks {self.defender_name}! {roll_text} HIT! Damage: {self.damage}"
    
    elif self.result == HitResult.MISS:
        if perspective == "attacker_is_player":
            return f"Player attacks {self.defender_name}... {roll_text} MISS!"
        elif perspective == "defender_is_player":
            return f"{self.attacker_name} attacks player... {roll_text} MISS!"
        else:
            return f"{self.attacker_name} attacks {self.defender_name}... {roll_text} MISS!"
    
    elif self.result == HitResult.CRITICAL_FAIL:
        if perspective == "attacker_is_player":
            return f"Player attempts to strike {self.defender_name}... {roll_text} CRITICAL MISS!"
        elif perspective == "defender_is_player":
            return f"{self.attacker_name} attempts to strike player... {roll_text} CRITICAL MISS!"
        else:
            return f"{self.attacker_name} attempts to strike {self.defender_name}... {roll_text} CRITICAL MISS!"
    
    # Fallback
    return f"Combat: {self.attacker_name} vs {self.defender_name}"
```

```python
def render_combat_messages(self, game):
    """
    Render three categorized combat message lines.
    
    Reads from game.combat_message_log which is a dict with keys:
      - "player_actions": list of str (most recent last)
      - "against_player": list of str (most recent last)
      - "observable": list of str (most recent last)
    
    Each line shows the most recent message in its category.
    If no message exists, show an empty line.
    """
    combat_msgs = getattr(game, "combat_message_log", None)
    if combat_msgs is None:
        # Fallback: render nothing if attribute doesn't exist
        return
    
    # Line 1: Player Actions (ui_y + 2)
    player_actions = combat_msgs.get("player_actions", [])
    if player_actions:
        msg = player_actions[-1]
        prefix = "[YOU] "
        self._render_text(0, self.ui_y + 2, (prefix + msg)[:self.console_width], COLORS['hp_high'])
    else:
        self._render_text(0, self.ui_y + 2, "[YOU] "[:self.console_width], COLORS['text_dim'])
    
    # Line 2: Actions Against Player (ui_y + 3)
    against_player = combat_msgs.get("against_player", [])
    if against_player:
        msg = against_player[-1]
        prefix = "[ATK] "
        self._render_text(0, self.ui_y + 3, (prefix + msg)[:self.console_width], COLORS['hp_low'])
    else:
        self._render_text(0, self.ui_y + 3, "[ATK] "[:self.console_width], COLORS['text_dim'])
    
    # Line 3: Observable/Ambient (ui_y + 4)
    observable = combat_msgs.get("observable", [])
    if observable:
        msg = observable[-1]
        prefix = "[OBS] "
        self._render_text(0, self.ui_y + 4, (prefix + msg)[:self.console_width], COLORS['text'])
    else:
        self._render_text(0, self.ui_y + 4, "[OBS] "[:self.console_width], COLORS['text_dim'])
```

```python
def attack(self, attacker: Entity, defender: Entity):
    event = CombatResolver.resolve_attack(attacker, defender)
    self.combat_log.add_event(event)
    
    # Determine perspective for message routing
    if attacker is self.player:
        perspective = "attacker_is_player"
    elif defender is self.player:
        perspective = "defender_is_player"
    else:
        perspective = "neutral"
    
    # Route to categorized combat message system
    self.add_combat_message(event, attacker, defender, perspective)
    
    # Also add to general message_log for backward compatibility
    self.add_message(str(event))
    
    if event.result in (HitResult.HIT, HitResult.CRITICAL):
        defender.hp -= event.damage
        if defender.hp <= 0:
            defender.hp = 0
            self.on_kill(attacker, defender)
```

### 4.7 `Game.add_combat_message()` — New method (after line 2839)

```python
def add_combat_message(self, event: CombatEvent, attacker: Entity, defender: Entity, perspective: str):
    """
    Route a combat event to the appropriate categorized message line.
    
    Args:
        event: The CombatEvent that occurred
        attacker: The attacking Entity
        defender: The defending Entity  
        perspective: One of "attacker_is_player", "defender_is_player", "neutral"
    """
    # Initialize combat_message_log if it doesn't exist
    if not hasattr(self, "combat_message_log"):
        self.combat_message_log = {
            "player_actions": [],
            "against_player": [],
            "observable": [],
        }
    
    # Generate the perspective-aware message string
    msg = str(event)  # Uses default neutral perspective for message_log
    
    # Determine which category this belongs to
    if attacker is self.player:
        # Player is attacking — goes to "player_actions" line
        self.combat_message_log["player_actions"].append(str(event.__class__.__str__(event, "attacker_is_player")))
    elif defender is self.player:
        # Player is being attacked — goes to "against_player" line
        self.combat_message_log["against_player"].append(str(event.__class__.__str__(event, "defender_is_player")))
    else:
        # Neither is player — only show if attacker is visible in FOV
        if self.fov is not None and 0 <= attacker.x < self.fov.shape[0] and 0 <= attacker.y < self.fov.shape[1]:
            if self.fov[attacker.x, attacker.y]:
                self.combat_message_log["observable"].append(str(event.__class__.__str__(event, "neutral")))
    
    # Trim each category to max 20 entries to prevent unbounded growth
    for key in self.combat_message_log:
        if len(self.combat_message_log[key]) > 20:
            self.combat_message_log[key] = self.combat_message_log[key][-20:]
```

**IMPORTANT NOTE ON IMPLEMENTATION:** The pseudocode above uses `event.__class__.__str__(event, perspective)` which is awkward. The actual implementation should call the `__str__` method directly with the perspective argument. Since Python's `str()` doesn't pass extra arguments to `__str__`, the coder must call `event.__str__(perspective)` directly, OR use a helper function. The coder should implement it as:

```python
# In add_combat_message, use this pattern:
from darkdelve import CombatEvent  # already in same file, just call directly

def _combat_event_str(event: CombatEvent, perspective: str) -> str:
    """Helper to call CombatEvent.__str__ with perspective argument."""
    return event.__str__(perspective)
```

So the actual `add_combat_message` body should be:

```python
def add_combat_message(self, event: CombatEvent, attacker: Entity, defender: Entity, perspective: str):
    if not hasattr(self, "combat_message_log"):
        self.combat_message_log = {
            "player_actions": [],
            "against_player": [],
            "observable": [],
        }
    
    # Generate the perspective-aware message
    msg = event.__str__(perspective)
    
    if attacker is self.player:
        self.combat_message_log["player_actions"].append(msg)
    elif defender is self.player:
        self.combat_message_log["against_player"].append(msg)
    else:
        # Neither is player — only show if attacker is visible in FOV
        if self.fov is not None:
            if (0 <= attacker.x < self.fov.shape[0] and 
                0 <= attacker.y < self.fov.shape[1]):
                if bool(self.fov[attacker.x, attacker.y]):
                    self.combat_message_log["observable"].append(msg)
    
    # Trim to max 20 entries per category
    for key in self.combat_message_log:
        if len(self.combat_message_log[key]) > 20:
            self.combat_message_log[key] = self.combat_message_log[key][-20:]
```

## 7. Integration Notes

### 7.1 How it integrates with existing systems

1. **`CombatEvent.__str__()`**: The existing method signature changes from `def __str__(self) -> str` to `def __str__(self, perspective: str = "neutral") -> str`. The default value `"neutral"` ensures backward compatibility — all existing calls to `str(event)` or `f"{event}"` will continue to work with neutral (third-person) language.

2. **`Game.attack()`**: Previously called `self.add_message(str(event))`. Now calls both `self.add_combat_message(event, attacker, defender, perspective)` AND `self.add_message(str(event))` for backward compatibility. The `message_log` is still populated so playtest infrastructure that reads it continues to work.

3. **`UI.render_messages()`**: Replaced by `UI.render_combat_messages()`. The old method is removed. The new method reads from `game.combat_message_log` instead of `game.message_log`.

4. **`UI.render_ui()`**: Updated to call `self.render_combat_messages(game)` instead of `self.render_messages(game)`.

5. **`UI.render_combat_log()`**: Updated to render at `ui_y + 5, 6, 7` instead of `ui_y + 5, 6` to accommodate the 3 message lines above it.

6. **`UI.__init__()`**: Updated to reserve 8 rows instead of 7 for the UI area.

7. **`Game.__init__()`**: New attribute `self.combat_message_log` initialized as a dict with three empty lists.

### 7.2 Backward Compatibility

- `add_message()` is preserved for non-combat messages (level up, pickup, stairs, etc.)
- `message_log` still receives combat messages via `add_message(str(event))` in `attack()`
- Playtest infrastructure that reads `message_log` continues to work
- `CombatEvent.__str__()` default parameter ensures existing `str(event)` calls work

## 8. Risks and Mitigations

### 8.1 Risk: `CombatEvent.__str__()` signature change breaks existing callers

**Severity**: Medium  
**Likelihood**: Low  
**Mitigation**: The `perspective` parameter has a default value of `"neutral"`, so all existing callers that use `str(event)` will continue to work. The neutral perspective produces the same output as the current implementation.

### 8.2 Risk: UI layout overflow on small terminals

**Severity**: Medium  
**Likelihood**: Medium  
**Mitigation**: The `UI.__init__()` already clamps `ui_y` to `console_height - 8`. If the terminal is too small, the UI will overlap with the map area. This is acceptable because the existing code already has this behavior for 7 rows. The config default is 80×50 which provides ample space.

### 8.3 Risk: FOV check in `add_combat_message` could fail if FOV is not initialized

**Severity**: Low  
**Likelihood**: Low  
**Mitigation**: The code checks `if self.fov is not None` before accessing FOV data. If FOV is None, observable events are simply not added (safe default).

### 8.4 Risk: `combat_message_log` could grow unboundedly

**Severity**: Low  
**Likelihood**: Low  
**Mitigation**: Each category is trimmed to max 20 entries in `add_combat_message()`. This prevents memory issues during long play sessions.

### 8.5 Risk: Playtest tests that check `message_log` for combat messages

**Severity**: Low  
**Likelihood**: Low  
**Mitigation**: The `attack()` method still calls `self.add_message(str(event))` after `add_combat_message()`, so `message_log` still contains all combat messages for backward compatibility.

### 8.6 Risk: `event.__str__(perspective)` syntax is unusual

**Severity**: Low  
**Likelihood**: Low (developer confusion)  
**Mitigation**: Documented clearly in the pseudocode. Python's `str()` builtin doesn't pass extra arguments to `__str__`, so direct method call is necessary. This is a standard Python pattern.
