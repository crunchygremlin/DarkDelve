def generate_floor1_description(self, game, state):
    import time
    player_class = getattr(game.player, "player_class", "adventurer")
    player_hp = game.player.max_hp
    theme = getattr(game, "current_theme", None)
    theme_name = theme.name if theme else "dungeon entrance"
    theme_desc = theme.description if theme else "A cold stone dungeon"

    prompt = f"""You are a dungeon master describing the entrance level of a roguelike dungeon.

Player: {player_class} (HP: {player_hp})
Theme: {theme_name} - {theme_desc}
Depth: 1 (entrance level)

Describe this floor in 3-5 sentences. Include:
- The entrance area (top of the map)
- Corridors and side rooms
- Monster types present (e.g., spiders, guards, rats, bats, trolls)
- Special features (dens, patrol routes, corpses, stairs)
- The stairs down (bottom of the map)

Use concrete monster names: spider, giant_spider, guard, dungeon_guard, sergeant, rat, cave_rat, bat, cave_bat, troll, troll_scavenger, fungal_creeper.
Use concrete feature names: corridor, den, patrol, corpse, stairs, web, nest.

Return ONLY the description text, no JSON, no markdown."""

    cache_key = f"floor1_desc_{theme_name}_{player_class}"

    if self.llm_config.get("cache_enabled", True):
        cached = self.cache.get(cache_key, prompt)
        if cached:
            return cached

    try:
        result = self.ollama.generate(
            prompt,
            temperature=self.llm_config.get("temperature", 0.7),
            top_p=self.llm_config.get("top_p", 0.9),
            num_predict=self.llm_config.get("num_predict", 512),
            timeout=self.llm_config.get("timeout", 30)
        )

        if result:
            if isinstance(result, dict):
                text = result.get("response", result.get("text", str(result)))
            else:
                text = str(result)

            text = text.strip().strip('"').strip('`').strip()

            if self.llm_config.get("cache_enabled", True):
                self.cache.set(cache_key, prompt, text, self.llm_config.get("model", "unknown"))

            return text

    except Exception as e:
        if hasattr(game, "llm_logger") and game.llm_logger:
            from src.domain.value_objects.llm_logging import LLMCallLog
            game.llm_logger.log_call(LLMCallLog(
                call_id=f"floor1_desc_error_{time.time()}",
                timestamp=time.time(),
                context="floor1_description",
                entity_id=None,
                prompt_summary=prompt[:200],
                response_summary="",
                latency_ms=0, tokens_used=0, success=False,
                error=str(e),
                turn_number=getattr(state, "turn", 0),
                call_type="floor1_description_error",
            ))

    return ""
```

"""Parser for LLM-generated floor 1 descriptions.

Extracts entity and feature hints from natural language descriptions
and applies them to Floor1Data to influence spawning.
"""

import re
import random
from typing import Dict, Any
from src.application.services.floor1_generator import Floor1Data, Den, PatrolRoute


# Keyword to creature type mapping (sorted by length for greedy matching)
MONSTER_KEYWORDS = {
    "giant spider": "giant_spider",
    "spider queen": "spider_queen",
    "spider": "giant_spider",
    "web": "giant_spider",
    "dungeon guard": "dungeon_guard",
    "sergeants": "guard_sergeant",
    "sergeant": "guard_sergeant",
    "patrol": "dungeon_guard",
    "guard": "dungeon_guard",
    "cave rat": "cave_rat",
    "rat king": "rat_king",
    "nest": "cave_rat",
    "rat": "cave_rat",
    "cave bat": "cave_bat",
    "bat": "cave_bat",
    "troll scavenger": "troll_scavenger",
    "troll": "troll_scavenger",
    "fungal creeper": "fungal_creeper",
    "mushroom": "fungal_creeper",
    "fungus": "fungal_creeper",
    "creeper": "fungal_creeper",
}

FEATURE_KEYWORDS = {
    "corpses": "corpse",
    "remains": "corpse",
    "bones": "corpse",
    "corpse": "corpse",
    "web": "den",
    "nest": "den",
    "den": "den",
    "stairs": "stairs",
    "passage": "corridor",
    "corridor": "corridor",
    "watch": "patrol",
    "patrol": "patrol",
}


def parse_floor1_description(description, floor1_data, dungeon_map, config):
    """Parse an LLM description and modify Floor1Data based on extracted hints."""
    if not description or not description.strip():
        return floor1_data

    desc_lower = description.lower()
    monster_counts = _extract_monster_counts(desc_lower)
    feature_counts = _extract_feature_counts(desc_lower)

    _add_dens_from_description(floor1_data, monster_counts, dungeon_map, config)
    _add_roaming_from_description(floor1_data, monster_counts, dungeon_map, config)
    _add_corpses_from_description(floor1_data, feature_counts, dungeon_map, config)
    _add_patrols_from_description(floor1_data, feature_counts, config)

    return floor1_data


def _extract_monster_counts(desc_lower):
    """Count monster keyword occurrences in description."""
    counts = {}
    sorted_keywords = sorted(MONSTER_KEYWORDS.keys(), key=len, reverse=True)
    for keyword in sorted_keywords:
        creature_type = MONSTER_KEYWORDS[keyword]
        pattern = r'\b' + re.escape(keyword) + r'\b'
        occurrences = len(re.findall(pattern, desc_lower))
        if occurrences > 0:
            counts[creature_type] = counts.get(creature_type, 0) + occurrences
    return counts


def _extract_feature_counts(desc_lower):
    """Count feature keyword occurrences in description."""
    counts = {}
    for keyword, feature_type in FEATURE_KEYWORDS.items():
        pattern = r'\b' + re.escape(keyword) + r'\b'
        occurrences = len(re.findall(pattern, desc_lower))
        if occurrences > 0:
            counts[feature_type] = counts.get(feature_type, 0) + occurrences
    return counts


def _add_dens_from_description(floor1_data, monster_counts, dungeon_map, config):
    """Add extra dens based on monster mentions (2+ mentions = extra den)."""
    width = config["dungeon"]["width"]
    height = config["dungeon"]["height"]
    existing_path_set = set(floor1_data.main_path)
    den_creatures = {"giant_spider", "cave_rat", "cave_bat", "fungal_creeper"}

    for creature_type, count in monster_counts.items():
        if creature_type not in den_creatures or count < 2:
            continue
        for _ in range(50):
            x = random.randint(3, width - 6)
            y = random.randint(3, height - 6)
            if not dungeon_map[x, y] and (x, y) not in existing_path_set:
                too_close = any(
                    abs(x - d.center[0]) + abs(y - d.center[1]) < 8
                    for d in floor1_data.dens
                )
                if not too_close:
                    room_w = random.randint(3, 5)
                    room_h = random.randint(3, 4)
                    for rx in range(x, min(x + room_w, width - 1)):
                        for ry in range(y, min(y + room_h, height - 1)):
                            dungeon_map[rx, ry] = False
                    floor1_data.dens.append(Den(
                        center=(x + room_w // 2, y + room_h // 2),
                        radius=max(room_w, room_h) // 2,
                        creature_type=creature_type,
                        count=random.randint(2, 4),
                    ))
                    break


def _add_roaming_from_description(floor1_data, monster_counts, dungeon_map, config):
    """Add extra roaming spawns for mentioned monster types."""
    width = config["dungeon"]["width"]
    height = config["dungeon"]["height"]
    existing_path_set = set(floor1_data.main_path)
    roaming_creatures = {"troll_scavenger", "cave_bat", "fungal_creeper", "cave_rat"}

    for creature_type, count in monster_counts.items():
        if creature_type not in roaming_creatures:
            continue
        extra = min(count - 1, 2)
        for _ in range(extra):
            for _ in range(50):
                x = random.randint(1, width - 2)
                y = random.randint(1, height - 2)
                if not dungeon_map[x, y] and (x, y) not in existing_path_set:
                    min_dist = min(abs(x - px) + abs(y - py) for px, py in floor1_data.main_path)
                    if min_dist > 3:
                        floor1_data.roaming_spawns.append((x, y, creature_type))
                        break


def _add_corpses_from_description(floor1_data, feature_counts, dungeon_map, config):
    """Add extra corpse positions if corpses are mentioned."""
    if "corpse" not in feature_counts:
        return
    width = config["dungeon"]["width"]
    height = config["dungeon"]["height"]
    for _ in range(min(feature_counts["corpse"], 3)):
        for _ in range(50):
            x = random.randint(1, width - 2)
            y = random.randint(1, height - 2)
            if not dungeon_map[x, y] and (x, y) not in floor1_data.corpses:
                floor1_data.corpses.append((x, y))
                break


def _add_patrols_from_description(floor1_data, feature_counts, config):
    """Add extra patrol routes if patrols are mentioned."""
    if "patrol" not in feature_counts or len(floor1_data.main_path) < 10:
        return
    segment_size = len(floor1_data.main_path) // (len(floor1_data.patrol_routes) + 1)
    waypoints = [
        floor1_data.main_path[idx]
        for idx in range(0, len(floor1_data.main_path), max(1, segment_size // 3))
    ]
    if len(waypoints) >= 2:
        floor1_data.patrol_routes.append(PatrolRoute(
            waypoints=waypoints,
            guard_types=["dungeon_guard"],
        ))
```

def _generate_floor1(self):
    """Generate floor 1 with LLM description influencing entity placement."""
    import time
    from src.application.services.floor1_generator import Floor1Generator
    from src.application.services.floor1_spawner import Floor1Spawner
    from src.application.services.floor1_description_parser import parse_floor1_description

    # Generate procedural base
    floor1_gen = Floor1Generator(self.config)
    floor1_data = floor1_gen.generate()

    # LLM Description Step
    llm_description = ""
    if self.dm_enabled and self.content_generator:
        try:
            llm_description = self.content_generator.generate_floor1_description(self, self.state)
        except Exception:
            llm_description = ""

    # Parse description and modify floor1_data
    if llm_description:
        try:
            floor1_data = parse_floor1_description(
                description=llm_description,
                floor1_data=floor1_data,
                dungeon_map=floor1_data.dungeon_map,
                config=self.config,
            )
        except Exception:
            pass  # Use unmodified procedural data

    # Apply map
    self.dungeon_map = floor1_data.dungeon_map
    self.stair_down_pos = floor1_data.stair_down
    self.stair_up_pos = floor1_data.entrance
    self.player.x, self.player.y = floor1_data.entrance
    self.player.home_position = floor1_data.entrance

    # Spawn entities
    spawner = Floor1Spawner(self.player, self.config, self.dungeon_map)
    self.entities = [self.player]
    self.entities.extend(spawner.spawn_all(floor1_data, None))

    # Generate minimal items
    items = self.content_generator.generate_items("martial", 5) if self.content_generator else []
    for item in items:
        for _ in range(20):
            x = random.randint(1, self.dungeon_map.shape[0] - 2)
            y = random.randint(1, self.dungeon_map.shape[1] - 2)
            main_path_set = set(floor1_data.main_path)
            if (not self.dungeon_map[x, y]
                    and not any(e.x == x and e.y == y for e in self.entities)
                    and (x, y) not in main_path_set):
                entity = Entity(
                    x=x, y=y, char=item.symbol, color=COLORS['item'],
                    name=item.name, blocks=False,
                    hp=1, max_hp=1, power=0, defense=0,
                    speed=0, intel_tier=0, is_commander=False,
                )
                entity.item = item
                self.entities.append(entity)
                break

    # Register AI agents
    for entity in self.entities:
        if entity is self.player:
            continue
        if getattr(entity, 'is_commander', False):
            agent = CommanderAgent(entity, home_position=(entity.x, entity.y))
        else:
            agent = RandomAgent(entity, agent_type=AgentType.MONSTER)
        self.agent_manager.register_agent(agent)

    # Request behavior scripts
    if self.dm_enabled and self.config.get('dungeon_master', {}).get('enable_behavior_generation', True):
        for entity in self.entities:
            if entity is self.player:
                continue
            if not hasattr(entity, 'behavior_script'):
                perception = self._get_perception_for_entity(entity)
                self.llm_request_queue.put({
                    'type': 'behavior',
                    'entity_id': entity.id,
                    'mob_type': getattr(entity, 'mob_type', 'default'),
                    'perception': perception.as_dict() if perception else {},
                    'social_context': '',
                    'valid_conditions': ['can_see_player', 'can_hear_player', 'health_below', 'in_combat'],
                    'valid_actions': ['attack', 'flee', 'patrol', 'wait'],
                    'turn_number': self.turn,
                    'prompt_summary': f'Behavior for {entity.name}',
                })

    # Initialize energy system
    self.energy_system = EnergySystem()
    for entity in self.entities:
        initial_energy = 100 if entity is self.player else 0
        self.energy_system.add_entity(entity, initial_energy=initial_energy)

    # Initialize FOV
    self.fov = self.fov_system.compute(self.dungeon_map, self.player.x, self.player.y)
    self.explored = self.fov_system.explored.copy()

    # Log the description
    if llm_description:
        self.add_message(f"The dungeon master whispers: {llm_description[:100]}...")
        if hasattr(self, 'llm_logger') and self.llm_logger:
            from src.domain.value_objects.llm_logging import LLMCallLog
            self.llm_logger.log_call(LLMCallLog(
                call_id=f"floor1_desc_{time.time()}",
                timestamp=time.time(),
                context="floor1_description",
                entity_id=None,
                prompt_summary=f"Floor 1 description for {getattr(self.player, 'player_class', 'adventurer')}",
                response_summary=llm_description[:200],
                latency_ms=0, tokens_used=0, success=True,
                turn_number=self.turn, call_type='floor1_description',
            ))

    self.add_message("You enter the dungeon entrance. The air smells of damp stone and something rotten.")

## 7. Integration Notes

1. **ContentGenerator.generate_floor1_description** follows the same caching pattern as generate_level_theme and generate_items, but uses self.ollama.generate() (plain text) instead of self.ollama.generate_json() since the output is prose, not structured data.

2. **parse_floor1_description** is a pure function that modifies Floor1Data in-place and returns it. It respects map bounds, walkable tiles, and minimum distance from existing features.

3. **Fallback chain**: If LLM is unavailable, generate_floor1_description returns empty string, parse_floor1_description returns floor1_data unchanged, procedural generation is used as-is. Zero regression.

4. **Logging**: Uses the existing LLMLogger infrastructure. New call_type values are floor1_description (success) and floor1_description_error (failure). These appear in logs/llm_activity.json alongside existing behavior/level_design entries.

5. **Non-blocking**: The LLM call happens during _generate_floor1() which is already called during level initialization (before the game loop starts). It does not use llm_request_queue because we need the result immediately to influence spawning. The timeout config (default 30s) prevents indefinite blocking.

6. **Coordinate system**: The parser uses dungeon_map[x, y] (True=wall, False=floor) consistent with the existing Floor1Generator and Floor1Spawner code.

## 8. Risks and Mitigations

| Risk | Mitigation |
|------|-----------|
| LLM call blocks game startup | timeout parameter (default 30s); exception caught leads to fallback |
| LLM returns garbage/empty string | parse_floor1_description checks if not description.strip() then returns unchanged |
| Parsed entities overlap existing ones | _find_valid_position in Floor1Spawner handles overlap; parser checks distance from existing dens |
| Parser carves rooms in walls | Parser only carves where dungeon_map[x, y] is already floor or carves small rooms explicitly |
